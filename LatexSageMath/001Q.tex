var('y')
t=ZZ.random_element(999999)
set_random_seed(t)

m=[(-1)^(ZZ.random_element(1,3))*(QQ.random_element(8)) for i in range(6)]
b=[(-1)^(ZZ.random_element(1,3))*(ZZ.random_element(0,12)) for i in range(len(m))]

eqnline=[m[i]*x+b[i] for i in range(len(m))]

## Create a plot object
plotxyplane = [plot([],figsize=(4,4),gridlines="minor") for i in range(len(m))]
plotxyplanea = [plot([],figsize=(4,4),gridlines="minor")+plot(eqnline[i],(x,-20,20), rgbcolor=(196/256,0/256,100/256)) for i in range(len(m))]

## Set the plot parameters
for i in range(len(m)):
    plotxyplane[i].set_axes_range(-20,20,-20,20)
    plotxyplane[i].axes_labels(['\(x\)','\(y\)']) 
    plotxyplane[i].axes_label_color('grey')
    plotxyplane[i].save(r'plottemp%s.pdf'%(i))
    plotxyplanea[i].set_axes_range(-20,20,-20,20)
    plotxyplanea[i].axes_labels(['\(x\)','\(y\)']) 
    plotxyplanea[i].axes_label_color('grey')
    plotxyplanea[i].save(r'a-plottemp%s.pdf'%(i))
 
qoutput=""
aoutput=""

for i in range(len(m)):
    qoutput += r"\item $y=%s$ \newline \includegraphics[scale=0.7]{plottemp%s.pdf}" %(latex(eqnline[i]),i)
    aoutput += r"\item $y=%s$ \newline \includegraphics[scale=0.7]{a-plottemp%s.pdf} \newline Slope: $m=%s$, \(y\)-intercept=$%s$" %(latex(eqnline[i]),i,m[i],b[i])

\end{sagesilent}
