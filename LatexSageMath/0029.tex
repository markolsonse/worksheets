\begin{sagesilent}
t=ZZ.random_element(999999)
set_random_seed(t)
numberOfQuestions = 48

ak = []
bk = []
gcdk = []

while len(gcdk) < numberOfQuestions:
    a = (ZZ.random_element(2,101))
    b = (ZZ.random_element(2,101))
    if gcd(a,b) != 1:
        ak.append(a)
        bk.append(b)
        gcdk.append(gcd(a,b))

qoutput = ""
aoutput = ""

for i in range(len(gcdk)):
  qoutput += rf"\item \( \hcf({ak[i]}, {bk[i]})  \) "
  aoutput += rf"\item \cRed{{ \( \hcf( {ak[i]}, {bk[i]} ) = {gcdk[i]} \) }}"
\end{sagesilent}
